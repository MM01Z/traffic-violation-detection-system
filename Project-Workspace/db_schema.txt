USE traffic_violations;

    id INT AUTO_INCREMENT PRIMARY KEY,
    violation_datetime DATETIME,
    rider_image VARCHAR(255),
    plate_image VARCHAR(255),

    challan_sent BOOLEAN DEFAULT FALSE,
    challan_price INT
